5pub(crate) fn format_shell_profile_scan(report: &ShellProfileScanReport) -> String {
6    let mut lines = vec![format!("Shell profile scan: {}", report.home)];
7
8    let profiles_with_mutations = report
9        .profiles
10        .iter()
11        .filter(|profile| !profile.path_mutations.is_empty())
12        .collect::<Vec<_>>();
13    let unreadable_profiles = report
14        .profiles
15        .iter()
16        .filter(|profile| profile.exists && profile.is_file && !profile.readable)
17        .collect::<Vec<_>>();
18
19    if profiles_with_mutations.is_empty() && unreadable_profiles.is_empty() {
20        lines.push(String::new());
21        lines.push("No PATH changes found in readable shell profiles.".to_owned());
22        return finish_lines(lines);
23    }
24
25    if !profiles_with_mutations.is_empty() {
26        lines.push(String::new());
27        lines.push("PATH changes:".to_owned());
28        for profile in profiles_with_mutations {
29            append_profile_mutations(&mut lines, profile);
30        }
31    }
32
33    if !unreadable_profiles.is_empty() {
34        lines.push(String::new());
35        lines.push("Unreadable profiles:".to_owned());
36        lines.extend(
37            unreadable_profiles
38                .iter()
39                .map(|profile| format!("  {} ({})", profile.path, profile.shell)),
40        );
41    }
42
43    finish_lines(lines)
44}
45
46fn append_profile_mutations(lines: &mut Vec<String>, profile: &ShellProfile) {
47    lines.push(format!("  {} ({})", profile.path, profile.shell));
48    lines.extend(profile.path_mutations.iter().map(|mutation| {
49        format!(
50            "    line {}  {}  {}",
51            mutation.line, mutation.kind, mutation.text
52        )
53    }));
54}
