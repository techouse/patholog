9pub(crate) fn dumps_json(value: &Value) -> Result<String, serde_json::Error> {

38pub(crate) fn resolution_to_json(report: &ResolutionReport) -> Value {
39    let winner = report
40        .candidates
41        .iter()
42        .find(|candidate| candidate.wins)
43        .map(candidate_to_json);
44
45    json!({
46        "command": report.command,
47        "found": winner.is_some(),
48        "winner": winner,
49        "candidates": report.candidates.iter().map(candidate_to_json).collect::<Vec<_>>(),
50        "searched_directories": report.searched_directories,
51        "related_hints": report.related_hints.iter().map(|hint| {
52            json!({
53                "command": hint.command,
54                "paths": hint.paths,
55            })
56        }).collect::<Vec<_>>(),
57    })
58}

101pub(crate) fn apply_outcome_to_json(outcome: &ApplyOutcome) -> Value {
102    let mut value = apply_plan_to_json(&outcome.plan);
103    if let Some(object) = value.as_object_mut() {
104        object.insert("wrote".to_owned(), json!(outcome.wrote));
105        object.insert("backup_path".to_owned(), json!(outcome.backup_path));
106        object.insert("backup_created".to_owned(), json!(outcome.backup_created));
107    }
108    value
109}

136fn entry_to_json(entry: &PathEntry) -> Value {
137    json!({
138        "index": entry.index,
139        "raw": entry.raw,
140        "display": entry.display,
141        "comparison_key": entry.comparison_key,
142        "kind": entry_kind(entry),
143        "exists": entry.exists,
144        "is_dir": entry.is_dir,
145        "is_empty": entry.is_empty,
146    })
147}
148
149fn entry_kind(entry: &PathEntry) -> &'static str {
150    if entry.is_empty {
151        "empty"
152    } else if !entry.exists {
153        "missing"
154    } else if !entry.is_dir {
155        "not_directory"
156    } else {
157        "directory"
158    }
159}

171fn health_counts_to_json(report: &HealthReport) -> Value {
172    let mut counts = serde_json::Map::new();
173    for (kind, count) in &report.counts {
174        counts.insert(kind.as_str().to_owned(), json!(count));
175    }
176    Value::Object(counts)
177}
