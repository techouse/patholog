3/// Planned profile edit action.
4#[derive(Clone, Copy, Debug, Eq, PartialEq)]
5pub enum ApplyAction {
6    /// The target profile does not exist and would be created by a future mutating apply.
7    CreateProfile,
8    /// The target profile exists without a patholog managed block.
9    AppendBlock,
10    /// The target profile has one complete patholog managed block.
11    ReplaceBlock,
12}
13
14impl ApplyAction {
15    /// Stable action string for human and JSON output.
16    #[must_use]
17    pub fn as_str(self) -> &'static str {
18        match self {
19            Self::CreateProfile => "create_profile",
20            Self::AppendBlock => "append_block",
21            Self::ReplaceBlock => "replace_block",
22        }
23    }
24}

26/// Plan describing what an apply operation would write.
27#[derive(Clone, Debug, Eq, PartialEq)]
28pub struct ApplyPlan {
29    /// Shell syntax used by the planned block.
30    pub shell: ShellKind,
31    /// Target shell profile path.
32    pub profile_path: String,
33    /// Planned operation.
34    pub action: ApplyAction,
35    /// Existing managed block when replacing one.
36    pub existing_block: Option<String>,
37    /// Complete managed block that apply writes or would write.
38    pub planned_block: String,
39    /// Cleaned PATH value used to render the planned block.
40    pub cleaned_path: String,
41    /// Whether this run writes files.
42    pub would_write: bool,
43}

45/// Result of a mutating apply run.
